1. You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string.

Input: abcadeecfb

Output: abcdef" 

      function removeDuplicate(str, n)
    {
      var index = 0;
      for (var i = 0; i < n; i++)
      {
        var j;
        for (j = 0; j < i; j++)
        {
         if (str[i] == str[j])
         {
           break;
         }
        }
        if (j == i)
        {
          str[index++] = str[i];
        }
      }
         
     return str.join("").slice(str, index);
    }
   var str = "abcadeecfb".split("");
   var n = str.length;
   console.log(removeDuplicate(str, n));

2. You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

Input: abcadeecfb
Output: a=2 b=2 c=2 d=1 e=2 f=1 
      let SIZE = 26;
    function printCharWithFreq(str)
    {
        let n = str.length;
  
        let freq = new Array(SIZE);
        for(let i = 0; i < freq.length; i++)
        {
            freq[i] = 0;
        }
        for (let i = 0; i < n; i++)
            freq[str[i].charCodeAt(0) - 'a'.charCodeAt(0)]++;
  
        for (let i = 0; i < n; i++) {
  
            if (freq[str[i].charCodeAt(0) - 'a'.charCodeAt(0)] != 0) {
  
                document.write(str[i]);
                document.write(freq[str[i].charCodeAt(0) - 'a'.charCodeAt(0)] + " ");
  
                freq[str[i].charCodeAt(0) - 'a'.charCodeAt(0)] = 0;
            }
        }
    }
let str = "abcadeecfb";
printCharWithFreq(str);
